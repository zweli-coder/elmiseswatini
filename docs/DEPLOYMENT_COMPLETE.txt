// Delete button triggers modal
const handleDeleteClick = (pub) => {
  setDeleteConfirm(pub);

// Modal shows publication details
<div className="modal-overlay">
  <div className="modal-content">
    <h2>Delete Publication?</h2>
    <p className="publication-title">{deleteConfirm.title}</p>
    <p className="warning-text">This action cannot be undone.</p>
    <button onClick={confirmDelete}>Delete Permanently</button>
  </div>
</div>

// Confirmed delete calls API
const confirmDelete = async () => {
  const response = await fetch(
    `${API_ENDPOINT}/publications/${deleteConfirm.id}`,
    {
      method: 'DELETE',
      headers: { 'Authorization': `Bearer ${token}` }
    }
  );
  // Remove from UI
  setPublications(publications.filter(p => p.id !== deleteConfirm.id));
  setDeleteConfirm(null);
};
